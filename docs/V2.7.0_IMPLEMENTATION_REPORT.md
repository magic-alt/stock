## 🔲 补丁2: 策略模板抽象（设计完成）

### 核心概念

引入**策略模板协议**（`StrategyTemplate`），定义标准生命周期：
```python
class StrategyTemplate(Protocol):
    params: Dict[str, Any]
    def on_init(self) -> None: ...     # 初始化指标
    def on_start(self) -> None: ...    # 策略启动
    def on_bar(self, symbol: str, bar: pd.Series) -> None: ...  # 逐 bar 回调
    def on_stop(self) -> None: ...     # 策略停止
```

**优势**:
- 🔹 **简化策略开发**: 不需要理解 Backtrader 复杂机制
- 🔹 **框架无关**: 模板可适配到 Backtrader/自定义 Runner
- 🔹 **与 V2.6.0 事件总线天然集成**: 策略可直接 `events.put(Event(...))`
- 🔹 **兼容现有策略**: 保留 `StrategyModule` + Backtrader 策略

```python
from __future__ import annotations
from typing import Protocol, Dict, Any
import pandas as pd

class StrategyTemplate(Protocol):
    """Simplified strategy protocol decoupled from execution engine."""
    params: Dict[str, Any]
    def on_init(self) -> None: ...
    def on_start(self) -> None: ...
    def on_bar(self, symbol: str, bar: pd.Series) -> None: ...
    def on_stop(self) -> None: ...

class BacktraderAdapter:
    """Adapts StrategyTemplate to Backtrader.Strategy."""
    def __init__(self, template_cls: type[StrategyTemplate], **params: Any):
        self.template_cls = template_cls
        self.params = params
    
    def to_bt_strategy(self):
        import backtrader as bt
        tmpl_cls, params = self.template_cls, dict(self.params)
        
        class _BT(bt.Strategy):
            params = params
            def __init__(self):
                self._tmpl: StrategyTemplate = tmpl_cls()
                self._tmpl.params = dict(self.params.__dict__)
                self._names = {data._name or f"sym{i}": data for i, data in enumerate(self.datas)}
                self._tmpl.on_init()
            def start(self): self._tmpl.on_start()
            def next(self):
                for name, d in self._names.items():
                    bar = pd.Series({"open": d.open[0], "high": d.high[0], 
                                     "low": d.low[0], "close": d.close[0], 
                                     "volume": getattr(d, "volume", [0])[0]})
                    self._tmpl.on_bar(name, bar)
            def stop(self): self._tmpl.on_stop()
        return _BT
```

```python
from __future__ import annotations
import os
import pandas as pd
from src.core.events import Event, EventType
from src.backtest.analysis import pareto_front, save_heatmap

def make_pipeline_handlers(out_dir: str):
    """Create event handlers for pipeline visualization and persistence."""
    os.makedirs(out_dir, exist_ok=True)
    results_buffer: dict[str, list[dict]] = {}  # strategy -> results
    
    def on_metrics(ev: Event) -> None:
        """Collect metrics from each grid run."""
        strat = ev.data["strategy"]
        results_buffer.setdefault(strat, []).append({
            "params": ev.data["params"],
            **ev.data["metrics"]
        })
    
    def on_stage(ev: Event) -> None:
        """Handle pipeline stage events (start/done)."""
        if ev.data.get("stage") == "grid.done":
            strat = ev.data["strategy"]
            if strat in results_buffer:
                df = pd.DataFrame(results_buffer[strat])
                
                # 保存 CSV
                csv_path = os.path.join(out_dir, f"{strat}_grid_results.csv")
                df.to_csv(csv_path, index=False)
                
                # 生成热力图（复用现有 analysis.py 函数）
                # save_heatmap(module, df, out_dir)  # 需要 module 对象
                
                # 生成 Pareto 前沿（如果有足够数据）
                if len(df) > 10:
                    pf = pareto_front(df, ["sharpe", "total_return"], ["drawdown"])
                    pf_path = os.path.join(out_dir, f"{strat}_pareto.csv")
                    pf.to_csv(pf_path, index=False)
                
                # 清空缓冲
                results_buffer[strat] = []
    
    return [
        (EventType.METRICS_CALCULATED, on_metrics),
        (EventType.PIPELINE_STAGE, on_stage),
    ]
```

```python
from __future__ import annotations
from dataclasses import dataclass
from typing import Any, Dict, Optional
import pandas as pd
from src.core.events import Event, EventEngine, EventType
from src.core.gateway import TradeGateway

@dataclass(slots=True)
class _Position:
    size: int = 0
    avg_price: float = 0.0

class PaperGateway(TradeGateway):
    """
    Paper trading gateway with event-driven order matching.
    
    Features:
    - Fill orders at next bar's open price
    - Event publishing (order.sent/filled/cancelled)
    - In-memory account/position tracking
    - Configurable slippage
    """
    
    def __init__(self, events: EventEngine, slippage: float = 0.0):
        self.events = events
        self.slippage = slippage
        self._cash = 200_000.0
        self._positions: dict[str, _Position] = {}
        self._oid = 0
        self._pending: dict[int, dict] = {}
        self._last_prices: dict[str, float] = {}
    
    def _next_oid(self) -> int:
        self._oid += 1
        return self._oid
    
    def mark_price(self, symbol: str, price: float) -> None:
        """Update last known price for symbol."""
        self._last_prices[symbol] = price
    
    def send_order(self, symbol: str, side: str, size: int, 
                   price: Optional[float] = None, order_type: str = "market") -> Any:
        """
        Submit order (queued for next bar matching).
        
        Emits: EventType.ORDER_SENT
        """
        oid = self._next_oid()
        order = {
            "id": oid, "symbol": symbol, "side": side,
            "size": size, "type": order_type, "price": price
        }
        self._pending[oid] = order
        self.events.put(Event(EventType.ORDER_SENT, order))
        return oid
    
    def cancel_order(self, order_id: Any) -> None:
        """
        Cancel pending order.
        
        Emits: EventType.ORDER_CANCELLED
        """
        if order_id in self._pending:
            order = self._pending.pop(order_id)
            self.events.put(Event(EventType.ORDER_CANCELLED, order))
    
    def query_account(self) -> Dict[str, Any]:
        """Query account balance and equity."""
        equity = self._cash
        for sym, pos in self._positions.items():
            px = self._last_prices.get(sym, pos.avg_price)
            equity += pos.size * px
        return {"balance": self._cash, "equity": equity}
    
    def query_position(self, symbol: str) -> Dict[str, Any]:
        """Query position for symbol."""
        pos = self._positions.get(symbol, _Position())
        return {"symbol": symbol, "size": pos.size, "avg_price": pos.avg_price}
    
    def match_on_open(self, symbol: str, open_price: float) -> None:
        """
        Match all pending orders for symbol at bar's open price.
        
        Matching logic:
        - Fill at: open_price * (1 + slippage)
        - Update cash and positions
        - Emit: EventType.ORDER_FILLED
        
        Called by PaperRunner before strategy sees the bar.
        """
        fill_price = open_price * (1 + self.slippage)
        self._last_prices[symbol] = fill_price
        
        to_fill = [od for od in self._pending.values() if od["symbol"] == symbol]
        for od in to_fill:
            self._pending.pop(od["id"], None)
            side = od["side"]
            size = od["size"]
            value = fill_price * size
            
            pos = self._positions.setdefault(symbol, _Position())
            
            if side == "buy":
                self._cash -= value
                # 加权平均成本
                new_size = pos.size + size
                pos.avg_price = (pos.avg_price * pos.size + fill_price * size) / max(new_size, 1)
                pos.size = new_size
            elif side == "sell":
                actual_size = min(size, pos.size)
                self._cash += fill_price * actual_size
                pos.size = max(0, pos.size - actual_size)
                if pos.size == 0:
                    pos.avg_price = 0.0
            
            self.events.put(Event(EventType.ORDER_FILLED, {
                "id": od["id"],
                "symbol": symbol,
                "price": fill_price,
                "size": size,
                "side": side
            }))
```

**文件**: `src/core/paper_runner.py` (约 80 行)
```python
from __future__ import annotations
import pandas as pd
from typing import Dict
from src.strategy.template import StrategyTemplate
from src.core.events import EventEngine
from src.core.paper_gateway import PaperGateway

def run_paper(
    template: StrategyTemplate,
    data_map: Dict[str, pd.DataFrame],
    events: EventEngine,
    *,
    slippage: float = 0.0
) -> Dict[str, Any]:
    """
    Run strategy template with paper trading gateway.
    
    Workflow:
    1. Initialize strategy and gateway
    2. For each bar:
       a. Match pending orders at open price
       b. Call strategy.on_bar() with bar data
       c. Update last prices
    3. Strategy completes, return account state
    
    Args:
        template: Strategy implementing StrategyTemplate protocol
        data_map: {symbol: DataFrame} with OHLCV data
        events: EventEngine for order/trade events
        slippage: Slippage rate (default: 0.0)
    
    Returns:
        Final account state from gateway.query_account()
    """
    gw = PaperGateway(events, slippage=slippage)
    
    # Lifecycle: init -> start
    template.on_init()
    template.on_start()
    
    # Align all symbols' date index
    all_dates = sorted(set().union(*[set(df.index) for df in data_map.values()]))
    
    for dt in all_dates:
        # Step 1: Match pending orders at current bar's open
        for symbol, df in data_map.items():
            if dt in df.index:
                gw.match_on_open(symbol, float(df.loc[dt, "open"]))
        
        # Step 2: Let strategy see the bar and potentially submit orders
        for symbol, df in data_map.items():
            if dt in df.index:
                row = df.loc[dt]
                bar = pd.Series({
                    "open": float(row["open"]),
                    "high": float(row["high"]),
                    "low": float(row["low"]),
                    "close": float(row["close"]),
                    "volume": float(row["volume"])
                })
                template.on_bar(symbol, bar)
                gw.mark_price(symbol, float(row["close"]))
    
    # Lifecycle: stop
    template.on_stop()
    
    return gw.query_account()
```
