### 2. 数据格式标准化

**问题**:
- AKShare返回的数据列名是中文（"日期"、"开盘"等）
- Backtrader需要英文列名（"datetime"、"open"等）

**解决方案**:
```python
# 使用数据格式标准化（已在yfinance_source中实现）
def standardize_dataframe(df):
    column_mapping = {
        '日期': 'datetime', '开盘': 'open', '收盘': 'close',
        '最高': 'high', '最低': 'low', '成交量': 'volume'
    }
    df = df.rename(columns=column_mapping)
    df = df.set_index('datetime')
    return df
```

## 🎯 使用建议

### 推荐方式：直接使用新模块

```python
#!/usr/bin/env python
"""
使用V2.4.0模块化组件进行回测
"""
import backtrader as bt
from datetime import datetime, timedelta
from src.strategies import create_backtrader_strategy
from src.data_sources import DataSourceFactory
from src.backtest.report_generator import quick_report

# 1. 获取数据
source = DataSourceFactory.create('akshare')
end_date = datetime.now().strftime('%Y%m%d')
start_date = (datetime.now() - timedelta(days=365)).strftime('%Y%m%d')
df = source.get_stock_history('600519', start_date, end_date)

# 标准化列名
df = df.rename(columns={
    '日期': 'datetime', '开盘': 'open', '收盘': 'close',
    '最高': 'high', '最低': 'low', '成交量': 'volume'
})
df = df.set_index('datetime')

# 2. 创建Cerebro
cerebro = bt.Cerebro()

# 3. 添加数据
data_feed = bt.feeds.PandasData(dataname=df)
cerebro.adddata(data_feed)

# 4. 添加策略（使用模块化策略）
strategy_cls = create_backtrader_strategy('macd', fast=12, slow=26, signal=9)
cerebro.addstrategy(strategy_cls)

# 5. 设置参数
cerebro.broker.setcash(100000.0)
cerebro.broker.setcommission(commission=0.001)

# 6. 运行回测
print(f"初始资金: {cerebro.broker.getvalue():,.2f}")
results = cerebro.run()
print(f"最终资金: {cerebro.broker.getvalue():,.2f}")

# 7. 生成报告
# quick_report(df, output_dir='./reports')

# 8. 显示图表
cerebro.plot()
```
